## 四、程式碼填空　（共 27 分，每空 ~4 分）

以下是用 Array 實作的 Stack，請填入缺少的部分。

```cpp
class Stack {
private:
    int arr[1000];
    int topIndex;
public:
    Stack() : topIndex( _①_ ) {}        // 空 Stack，-1 代表無元素

    void push(int x) {
        arr[ _②_ ] = x;                 // topIndex 先加 1，再存入
    }

    int pop() {
        if ( _③_ ) throw "Underflow";
        return arr[ _④_ ];              // 回傳頂端值並讓 topIndex -1
    }

    int top() {
        return arr[ _⑤_ ];
    }

    bool isEmpty() { return topIndex == _⑥_; }
    int  size()    { return topIndex +  _⑦_; }
};
```
